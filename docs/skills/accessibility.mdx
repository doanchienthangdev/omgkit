## Keyboard Navigation
```tsx
function handleKeyDown(e: KeyboardEvent) {
  if (e.key === 'Enter' || e.key === ' ') {
    activate();
  }
  if (e.key === 'Escape') {
    close();
  }
}
```

## Focus Management
```tsx
const ref = useRef<HTMLButtonElement>(null);

useEffect(() => {
  if (isOpen) ref.current?.focus();
}, [isOpen]);
```
